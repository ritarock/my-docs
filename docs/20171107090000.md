date: 20171107090000
tags: ['JavaScript']

```javascript
Promise.resolve().then(function() {
    return new Promise(function(fulfilled, rejected) {
        //処理1
        fulfilled();
    });
}).then(function() {
    return new Promise(function(fulfilled, rejected) {
        //処理2
        fulfilled();
    });
}).then(function() {
    return new Promise(function(fulfilled, rejected) {
        //処理3
    });
})
```
